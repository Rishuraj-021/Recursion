#include <iostream>
using namespace std;


int fact(int n) {
    if (n <= 1)
        return 1;
    else
        return n * fact(n - 1);
}

int main() {
    int number;

    cout << "Enter a number: ";
    cin >> number;

    cout << "Factorial of " << number << " is " << fact(number) << endl;

    return 0;
}
